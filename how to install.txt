----- хостинг ---
1. командная строка: ssh root@82.148.29.68
2. посмотреть что есть: команда ls
3. обновить всё: команда sudo apt update

4. удалить всё: 
sudo rm -rf Williams-test

--- установка программ:
1. sudo apt install git (установка git)
2. git --version (проверка, установлен ли git)
3. git clone https://github.com/IvanPazhgin/Williams-fractal-test-React_Node.git (копируем наш код)
4. ls (смотрим что есть на сервере)
5. идем в папку нашего проекта: cd "folder"

---- устанавливаем nodejs. 
инструкция: https://www.digitalocean.com/community/tutorials/how-to-install-node-js-on-ubuntu-22-04
// 6. sudo apt -y install curl dirmngr apt-transport-https lsb-release ca-certificates (этого нет в инструкии выше)

https://techviewleo.com/how-to-install-node-js-18-lts-on-ubuntu/
//6. curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
// 7. sudo apt-get install nodejs
7. sudo apt install nodejs
8. проверяем: node -v
9. sudo apt install npm
10. проверяем: npm -v
11. npm install (устанавливаем все библиотеки)
12. npm run client:install (установка пакетов для клиента)
13. npm run client:build

----- настройка файла config
14. cd config (заходим в папку)
15. nano production.json
16. меняем порт на 80
17. меняем baseUrl на домен или ip сервера
18. ctrl + O = сохранить. ctrl + X = выйти

---- запуск
19. npm run start

--- установка pm2 в папке проекта (для постоянно работы сервера)
20. sudo npm install -g pm2

---- запуск pm2
21. pm2 start npm -- start
--- остановить процесс
22. pm2 stop 0

--- при обновлении резила необходимо удалить процесс
23. pm2 delete 0
